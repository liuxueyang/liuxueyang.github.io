Hdu 4825 - Xor Sum

這是 2014 年百度之星資格賽題目。

給定一個包含 N 個數字的集合，發起 N 次詢問，每次詢問中給一個數字 S，要求在集合中找到一個數字 K，使得 K 和 S 的異或結果最大。

把數字的二進制表示放進字典樹裏面，對於數字 S，從數字 S 的最高二進制位開始，從字典樹樹根依次尋找儘量和當前 S 的二進制位不同的節點，最終形成的數字一定是最大的。

這是一個貪心的思路。唯一需要注意的就是，爲了處理方便，應該把每個數字的 32 位二進制位全部插入字典樹，這樣就避免了數字長短導致的問題。

開始寫的時候是把數字轉化成二進制字符串，然後再插入字典樹的。這樣就增加了字符串的操作，所以超時了： 

// Trie Tree
const int ALPHASIZE = 2;

class TrieNode {
public:
  TrieNode()
    :wordEnd{false}, cnt{1}
  {
    REP(i, ALPHASIZE) { child[i] = nullptr; }
  }
  ~TrieNode() {}
  void insert(string word) {
    int i;
    TrieNode *cur = this;
    for (auto x:word) {
      i = x - '0';
      if (nullptr == cur->child[i]) {
        cur->child[i] = new TrieNode();
        cur = cur->child[i];
      } else {
        cur = cur->child[i];
        ++cur->cnt;
      }
    }
    cur->wordEnd = true;
  }
  bool search(string word) {
    int i;
    TrieNode *cur = this;
    for (auto x:word) {
      i = x - '0';
      if (nullptr == cur->child[i]) {
        return false;
      }
      cur = cur->child[i];
    }
    return cur->wordEnd;
  }
  int cnt_words(string word) {
    int i; TrieNode *cur = this;
    for(auto x:word) {
      i = x - '0';
      if (nullptr == cur->child[i]) return 0;
      cur = cur->child[i];
    }
    return cur->cnt;
  }
  LL find_xor_num(string num) {
    LL res = 0;
    TrieNode *cur = this;
    // PR1(num);
    REPV(i, num.size()) {
      if (num[num.size()-1-i] == '0') {
        if (cur->child[1] != nullptr) { res += (1<<i); cur = cur->child[1]; }
        else cur = cur->child[0];
      } else {
        // PR1(cur->child[0]);
        if (cur->child[0] != nullptr) cur = cur->child[0];
        else { res += (1<<i); cur = cur->child[1]; }
      }
      // PR2(i, res);
    }
    return res;
  }
  TrieNode *child[ALPHASIZE];
  bool wordEnd;
  int cnt; // number of words with the prefix
};



string bin_num(LL x) {
  string s;
  while (x) {
    if (x&1) s += '1';
    else s += '0';
    x >>= 1;
  }
  reverse(s.begin(), s.end());
  while (s.size() < 33) s = "0" + s;
  return s;
}

// ==================================================


int main( void ) {

#ifdef DEBUG
  freopen("4825.in", "r", stdin);
#endif

  int T; cin >> T;
  FOR(ii, 1, T+1) {
    cout<<"Case #"<<ii<<":"<<endl;
    TrieNode tree;
    int N, M; cin >> N >> M;
    string s;
    LL t;
    REP(jj, N) {
      cin >> t;
      tree.insert(bin_num(t));
    }
    REP(jj, M) {
      cin >> t; s = bin_num(t);
      LL r = tree.find_xor_num(s);
      PRLN(r);
    }
  }

  return 0;
}

然後稍微優化了一下，把字符串操作都避免。插入的時候直接用位運算從最高位開始取二進制位就可以了，然後查找的時候同理。然後這題就可以過了。 

// Trie Tree
const int ALPHASIZE = 2;

class TrieNode {
public:
  TrieNode()
    :wordEnd{false}, cnt{1}
  {
    REP(i, ALPHASIZE) { child[i] = nullptr; }
  }
  ~TrieNode() {}
  void insert(int num) {
    int i;
    TrieNode *cur = this;
    LL tmp = 1LL<<32;
    REPV(j, 33) {
      tmp = 1 << j;
      i = (tmp & num) >> j;
      if (nullptr == cur->child[i]) {
        cur->child[i] = new TrieNode();
        cur = cur->child[i];
      } else {
        cur = cur->child[i];
        ++cur->cnt;
      }
    }
    cur->wordEnd = true;
  }
  bool search(string word) {
    int i;
    TrieNode *cur = this;
    for (auto x:word) {
      i = x - '0';
      if (nullptr == cur->child[i]) {
        return false;
      }
      cur = cur->child[i];
    }
    return cur->wordEnd;
  }
  int cnt_words(string word) {
    int i; TrieNode *cur = this;
    for(auto x:word) {
      i = x - '0';
      if (nullptr == cur->child[i]) return 0;
      cur = cur->child[i];
    }
    return cur->cnt;
  }
  LL find_xor_num(LL num) {
    LL res = 0;
    TrieNode *cur = this;

    int len=33;
    REPV(i, len) {
      if ((num & (1<<i)) == 0) {
        if (cur->child[1] != nullptr) { res += (1<<i); cur = cur->child[1]; }
        else cur = cur->child[0];
      } else {
        if (cur->child[0] != nullptr) cur = cur->child[0];
        else { res += (1<<i); cur = cur->child[1]; }
      }
    }
    return res;
  }
  TrieNode *child[ALPHASIZE];
  bool wordEnd;
  int cnt; // number of words with the prefix
};

// ==================================================


int main( void ) {

#ifdef DEBUG
  freopen("4825.in", "r", stdin);
#endif

  int T; cin >> T;
  FOR(ii, 1, T+1) {
    cout<<"Case #"<<ii<<":"<<endl;
    TrieNode tree;
    int N, M; cin >> N >> M;
    string s;
    LL t;
    REP(jj, N) {
      cin >> t;
      tree.insert(t);
    }
    REP(jj, M) {
      cin >> t;
      LL r = tree.find_xor_num(t);
      PRLN(r);
    }
  }

  return 0;
}

可見如果要依賴積累的模板的話也要把模板寫好一點，或者根據題目需求稍微優化一下。

另外，這種類型的字典樹叫做 01 字典樹，可以用來解決異或最大值的問題。
