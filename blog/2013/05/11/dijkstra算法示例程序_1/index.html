const int MAX = 20;
int S[MAX], dist[MAX], path[MAX], edge[MAX][MAX];
int n;
void Dijkstra(int v0) {
  int i, j, k;
  for (i = 0; i <n ; ++i) {
    dist[i] = edge[v0][i]; S[i] = 0;
    if (i != v0 && edge[v0][i] != MAXN) path[i] = v0;
    else path[i] = -1;
  }
  S[v0] = 1; dist[v0] = 0;
  for (i = 0; i < n - 1; ++i) {
    int Min = MAXN, u = v0;
    for (j = 0; j < n; ++j) {
      if (dist[j] < Min && S[j] == 0) {
        Min = dist[j]; u = j;
      }
    }
    S[u] = 1;
    for (j = 0; j < n; ++j) {
      if (S[j] == 0 && edge[u][j] != MAXN) {
        int tmp = edge[u][j] + dist[u];
        if (tmp < dist[j]) {
          dist[j] = tmp; path[j] = u;
        }
      }
    }
  }
}
