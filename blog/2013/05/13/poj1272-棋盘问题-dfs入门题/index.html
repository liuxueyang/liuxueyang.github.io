char m[MAX][MAX];int cnt, n, k, sum;
bool p[MAX];
bool judge(int i, int j) {
  if (p[j] == false && m[i][j] == '#') return true;
  else return false;
}
void dfs(int x) {
  if (sum == k) {cnt++; return;}
  if (x >= n) return;
  int i;
  for (i = 0; i < n; ++i) {
    if (judge(x, i)) {
      p[i] = true; sum++; dfs(x+1);
      p[i] = false; sum--;
    }
  }
  dfs(x + 1);
}
