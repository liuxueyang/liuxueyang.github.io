        for (int j = 0; j < 16; ++j)
        {
            if (A[i][j]!="")
            {
                so.first=i, so.second=j;
                bool mrk = false;
                for (int k = 0; k < 8; ++k)
                {
                    for (int h = 0; h < 16; ++h)
                    {
                        if (A[i][j]==B[k][h])
                        {
                            mrk = true;
                            to.first=k,to.second=h;
                            coll[so] = to; break;
                        }
                    }
                    if (mrk) break;
                }
            }
        }
    }
    for (int i = 0; i < sad.size(); i+=2)
    {
        int X, Y; tmp.clear();
        if (sad[i]>='A'&&sad[i]<='F') X = sad[i]- 'A'+10;
        else X = sad[i] - '0';
        if (sad[i+1]>='A'&&sad[i+1]<='F') Y = sad[i+1]- 'A'+10;
        else Y = sad[i+1] - '0';
        pair<int,int> hehe, TM;
        hehe.first = X, hehe.second = Y;
        TM = coll[hehe];
        int j = TM.first, k = TM.second;
        char XX, YY;
        XX = j+ '0';
        if (k >= 10) YY = k - 10 + 'A';
        else YY = k + '0';
        printf("%c%c", XX, YY);
    }
    printf("\n");

    return 0;
}
