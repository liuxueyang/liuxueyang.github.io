int V, c[111], w[111], f[55555], n;
int main(void)
{
    #ifndef ONLINE_JUDGE
    freopen("in.txt", "r", stdin);
    #endif // ONLINE_JUDGE
    int t; cin>>t;
    while (t--)
    {
        cin>>n; int sum=0;
        for (int i = 0; i < n; ++i)
        {
            cin>>c[i]; sum+=c[i];
        }
        V=sum/2;
        memset(f, 0, sizeof(f));
        for (int i = 0; i < n; ++i)
        {
            for (int v = V; v>=c[i]; --v)
            {
                f[v] = max(f[v], f[v-c[i]]+c[i]);
            }
        }
        cout<<fabs(f[V]-(sum-f[V]))<<endl;
    }
    return 0;
}
