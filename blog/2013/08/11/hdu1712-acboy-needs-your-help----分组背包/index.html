    {
        for (int i=1;i<=n;++i)
        {
            for (int j=1;j<=m;++j) scanf("%d",&a[i][j]);
        }
        memset(dp,0,sizeof(dp));
        for (int i=1;i<=n;++i)
        {
            for (int v=m;v>=0;--v)
            {
                for (int j=1;j<=v;++j)
                {
                    dp[v]=max(dp[v],dp[v-j]+a[i][j]);
                }
            }
        }
        printf("%d\n",dp[m]);
    }
    return 0;
