认真思考，发现，逆序的数字对一定有边相连。所以，题目要求没有边相连的最大的集合的点的个数，其实就是找原来的序列的最长上升子序列！

        touch[0] = -1;
        int tmp, top = 0;
        for (int i = 0; i < n; ++i) {
            scanf("%d", &tmp);
            if (tmp > touch[top]) {
                touch[++top] = tmp;
            } else {
                int pos = lower_bound(touch+1, touch+1+top, tmp) - touch;
                touch[pos] = tmp;
            }
        }
        printf("%d\n", top);
    }

}                /* ----------  end of function main  ---------- */
