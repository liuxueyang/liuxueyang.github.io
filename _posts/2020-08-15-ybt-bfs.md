  int z, x, y;
};

                      {0, 1, 0}, {0, 0, 1},
                      {0, -1, 0}, {0, 0, -1},
                      {1, 0, 0}, {-1, 0, 0},
};

            end = Point{i, j, k};
          } else if (g[i][j][k] == '#') {
            vis[i][j][k] = true;
          }
        }
      }
    }

    while (!que.empty()) {
      que.pop();
    }
    que.push(start);
    int res = -1;

      Point t = que.front();
      int z1 = t.z, x1 = t.x, y1 = t.y;
      int ori = a[z1][x1][y1];
      vis[z1][x1][y1] = true;

      for (int i = 0; i < 6; ++i) {
        int z2, x2, y2;

        x2 = x1+di[i][1];

            res = ori + 1;
            break;
          }
          vis[z2][x2][y2] = true;
          que.push(t1);
          a[z2][x2][y2] = ori + 1;
        }
      }
      if (res != -1) break;
    }
    if (res == -1) {
      cout << "Trapped!" << "\n";
    } else {
      cout << "Escaped in " << res << " minute(s)." << "\n";
    }
  }


  return 0;
}
