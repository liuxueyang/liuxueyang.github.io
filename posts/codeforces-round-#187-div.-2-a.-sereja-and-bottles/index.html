题目链接：http://codeforces.com/contest/315/problem/A 这到题目比赛的时候过了，赛后WA了……虽然很简单，但是还是没想太清楚


#include <iostream>
#include <cstdio>
#include <cstdlib>
using namespace std;
#define MAX 103
int a[MAX], b[MAX];
int main(void) {
    int n;
#ifndef ONLINE_JUDGE
    freopen("187a.in", "r", stdin);
#endif
    while (~scanf("%d", &n)) {
        for (int i = 0; i < n; ++i) scanf("%d%d", a+i, b+i);
        int cnt = 0;
        for (int i = 0; i < n; ++i) {
            bool flag = false;
            for (int j = 0; j < n; ++j) {
                if (i!=j && a[i] == b[j]) {
                    flag = true; break;
                }
            }
            if (!flag) cnt++;
        }
        printf("%d\n", cnt);

    }

    return 0;
}
