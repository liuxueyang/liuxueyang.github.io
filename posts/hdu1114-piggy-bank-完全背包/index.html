int f[11111],c[555],w[555], V, n;
void completepack(int c,int w) {
    for(int v=c;v<=V;++v) 
            f[v]=min(f[v],f[v-c]+w);
}

        memset(f,INF,sizeof(f)); f[0]=0;
        for (int i=0;i<n;++i) completepack(w[i],c[i]);
        if(f[V]==INF) cout<<"This is impossible.\n";
