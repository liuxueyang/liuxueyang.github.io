{

    ios::sync_with_stdio(false);
    int t; cin>>t;
    fi[0] = fi[1] = 1;
    for (int i = 2; i < 11111; ++i) fi[i] = (fi[i-1]+fi[i-2])%MOD;
    for (int i = 1; i <= t; ++i)
    {
        cin>>a; int len = strlen(a), cnt = 0, ans = 1;
        for (int j = 1; j < len; ++j)
        {
            if (a[j]=='e' && a[j-1]=='h')
            {
                ++j; cnt++;
            }
            else
            {
                ans = (ans*fi[cnt])%MOD; cnt = 0;
            }
        }
        ans = (ans*fi[cnt])%MOD;
        cout<<"Case "<<i<<": "<<ans<<endl;
    }

}
