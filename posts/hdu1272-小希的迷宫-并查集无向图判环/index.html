int parent[MAX];
bool flag;
void init() {
  for (int i = 1; i<=MAX; ++i) parent[i] = -1;
}
int find(int x) {
  int s;
  for (s = x; parent[s] != s; s = parent[s]) ;
  while (s != x) {
    int tmp = parent[x]; parent[x] = s; x = tmp;
  }
  return s;
}
void Union(int R1, int R2) {
  int r1 = find(R1), r2 = find(R2);
  if (r1 != r2) parent[r1] = r2;
  else flag = false;
}
