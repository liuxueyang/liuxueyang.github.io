题目链接：http://poj.org/problem?id=2407 题目大意： 　　这个题目就是欧拉函数的定义，求一个数字的欧拉函数。 题目思路： 　　用公式：φ(n) = n * (1-1/p1) * (1-1/p2) * ……  * (1-1/pk)。

#ifndef ONLINE_JUDGE

  LL n;
  while (~scanf("%lld", &n)){
    if (!n) break;
    LL ans = n;
    for (LL i = 2; i * i <= n; ++i){
      if (!(n%i)){
        ans = ans - ans / i;
        while (!(n%i)){
          n /= i;
        }
      }
    }
    if (n > 1) { ans = ans - ans / n; }
    printf("%lld\n", ans);
  }

  return 0;
}
