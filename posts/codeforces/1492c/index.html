找到 $t_i$ 在 $s$ 中的最左边的位置 $l_i$，这可以 $O(N)$ 求出来。同样可以求得每个 $t_i$ 在 $s$ 中最右边的位置 $r_i$。最终答案就是 $max(r_{i + 1} - l_{i})$

const int N = 200100;
int a[N], b[N];

void solve() {
  int n, m;
  string s, t;

  while (cin >> n >> m) {
    cin >> s >> t;

    int i = 0;

    For(j, 0, m) {
      while (i < n && s[i] != t[j])
        ++i;
      a[j] = i++;
    }

    i = n - 1;
    Rof(j, 0, m) {
      while (i >= 0 && s[i] != t[j])
        --i;
      b[j] = i--;
    }

    int ans = 1;

    For(i, 0, m - 1) {
      int tmp = b[i + 1] - a[i];
      ckmax(ans, tmp);
    }

    cout << ans << '\n';
  }
}
