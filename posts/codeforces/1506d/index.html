const int N = 200100;
int n, a[N];

void solve() {
  cin >> n;

  map<int, int> m;
  For1(i, 1, n) {
    cin >> a[i];
    m[a[i]]++;
  }

  int mx = 0;
  for (auto &[_, x] : m)
    ckmax(mx, x);

  if (2 * mx >= n) {
    cout << 2 * mx - n << '\n';
  } else {
    if (n & 1)
      cout << "1\n";
    else
      cout << "0\n";
  }
}
