void solve() {
  ll a, b, l;
  cin >> a >> b >> l;
  set<int> s;

  int a1 = 1;
  while (a1 <= l) {
    if (l % a1 != 0) {
      a1 *= a;
      continue;
    }

    int l1 = l / a1, b1 = 1;
    while (l1 % b1 == 0) {
      s.insert(l1 / b1);
      b1 *= b;
    }
    a1 *= a;
  }

  cout << SZ(s) << '\n';
}

int main(void) {
  std::ios::sync_with_stdio(false);
  cin.tie(NULL);
  cout.tie(NULL);

  int T = 1;
  cin >> T;

  while (T--) {
    solve();
  }

  return 0;
}
