先考虑每一个洞穴，为了要打败所有的怪物，考虑每个怪兽要求的起始等级，然后求最大值。接下来考虑所有的洞穴，每个洞穴的起始等级要求不一样，可以先从小到大排序，因为对于给定等级，判定能否通关只需要一次线性扫描，所以可以二分答案。除了二分解法，其实也可以按照单个洞穴的思路来考虑所有的洞穴。

const int N = 100010;
int t, n;
pair<ll, ll> b[N];
VI a[N];

bool check(ll x) {
  For(i, 0, n) {
    if (x < b[i].f1)
      return false;
    x += b[i].f2;
  }
  return true;
}

int main(void) {

      ll ma = a[i][0];
      For(j, 1, len) { ma = max(ma, ll(a[i][j] - j)); }
      b[i].f1 = ma;
      b[i].f2 = len;
    }

    sort(b, b + n);

    ll l = 1, r = 1e18, mid;
    while (l < r) {
      mid = (l + r) / 2;
      if (check(mid))
        r = mid;
      else
        l = mid + 1;
    }
