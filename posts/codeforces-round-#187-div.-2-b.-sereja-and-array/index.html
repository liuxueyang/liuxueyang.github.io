using namespace std;

typedef struct Po {
    int d, a;
}Po;
Po p[100000+10];
int main(void) {
#ifndef ONLINE_JUDGE
    freopen("187b.in", "r", stdin);
#endif
    int n, m;
    scanf("%d%d", &n, &m);
    for (int i = 1; i <= n; ++i) {
        scanf("%d", &p[i].a);
        p[i].d = 0;
    }
    int D = 0, t, s;
    for (int i = 0; i < m; ++i) {
        scanf("%d", &t);
        if (t == 1) {
            int st, en;
            scanf("%d%d", &st, &en);
            p[st].a = en;
            p[st].d = D;
        }
        else {
            scanf("%d", &s);
            if (t == 3) {
                printf("%d\n", p[s].a + D - p[s].d);
            }
            else if (t == 2) {
                D += s;
            }
        }
    }

    return 0;
}
