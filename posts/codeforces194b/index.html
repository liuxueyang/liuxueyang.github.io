int main(void)
{
    set<int> sadx, sady;
    int x[9] ,y[9] ;
    bool flag = true;
    for (int i=0;i<8;++i)
    {
        scanf("%d%d",&x[i],&y[i]);
        sadx.insert(x[i]); sady.insert(y[i]);
        for (int j=0;j<i;++j)
        {
            if (x[j]==x[i]&&y[j]==y[i])
            {
                flag = false;
            }
        }
    }
        if(!(sadx.size()==3&&sady.size()==3)) {
            flag = false;
        }
        int midx, midy; midx=*(++sadx.begin());
        midy=*(++sady.begin());
        for (int i=0;i<8;++i)
        {
            if(midx==x[i]&&midy==y[i])
            {
                flag=false; break;
            }
        }
        if(flag) printf("respectable\n");
        else printf("ugly\n");

        return 0;
}
