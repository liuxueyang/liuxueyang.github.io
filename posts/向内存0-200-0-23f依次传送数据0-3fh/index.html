只能用字节为单位传送了。

assume cs:sad
sad segment
start:
    mov ax, 20h
    mov ds, ax
    mov bx, 0
    mov cx, 40h
s:
    mov [bx], bl
    inc bl
    loop s

    mov ax, 4c00h
    int 21h
sad ends
end
