- [Masha and a Beautiful Tree](https://vjudge.net/problem/CodeForces-1741d)

经典的分治问题，记录区间的左右端点和区间范围即可。找到一个分支无解之后，及时退出。

const int N = 300010;
int t, n, a[N];
ll res;

void dfs(int l, int r, int vl, int vr) {
  if (l >= r)
    return;

  if (res == -1)
    return;

  if (l + 1 == r) {
    if (a[l] > a[r]) {
      res++;
      swap(a[l], a[r]);
    }
    if (PII{vl, vr} != PII{a[l], a[r]}) {
      res = -1;
    }
    return;
  }

  int mid = (l + r) / 2, vmid = (vl + vr) / 2;

  if (a[l] >= vl && a[l] <= vmid) {
    dfs(l, mid, vl, vmid);
    dfs(mid + 1, r, vmid + 1, vr);
  } else {
    res++;
    dfs(l, mid, vmid + 1, vr);
    dfs(mid + 1, r, vl, vmid);
  }
}
