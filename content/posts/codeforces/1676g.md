- [White-Black Balanced Subtrees](https://vjudge.net/problem/CodeForces-1676g)

一次 DFS 之后统计出各个节点子树的黑色节点和白色节点的个数。

const int N = 4010, M = 10010;
int h[N], w[N], b[N], t, n;
int e[M], ne[M], idx;
char color[N];

void Init() {
  idx = 0;
  memset(h, -1, sizeof h);
  memset(w, 0, sizeof w);
  memset(b, 0, sizeof b);
}

void Add(int a, int b) { e[idx] = b, ne[idx] = h[a], h[a] = idx++; }

void dfs(int x) {
  int w1 = 0, b1 = 0;

  for (int i = h[x]; i != -1; i = ne[i]) {
    int j = e[i];
    dfs(j);
    w1 += w[j];
    b1 += b[j];
  }

  if (color[x] == 'B')
    b1++;
  else
    w1++;

  w[x] = w1;
  b[x] = b1;
}

    For1(i, 1, n) {
      if (w[i] == b[i])
        res++;
    }
    cout << res << '\n';
  }

  return 0;
}
