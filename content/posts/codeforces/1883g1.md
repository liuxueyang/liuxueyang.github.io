- [Dances (Easy version)](https://vjudge.net/problem/CodeForces-1883g1)

对于每个 $a_i$，找到尽量小的 $b_j > a_i$，如果不存在，那么就应该移除。可以把 $a, b$ 排序，二分找 $b_j$。

const int N = 100010;
int n, m, a[N], b[N], t;

int solve(int x, int l) {
    int r = n, mid;
    if (l > r) return -1;

    while (l < r) {
        mid = (l + r) / 2;
        if (b[mid] > x) r = mid;
        else l = mid + 1;
    }
    if (b[r] > x) return r;
    return -1;
}

        int res = 0, l = 1;

                break;
            }
            l = pos + 1;
