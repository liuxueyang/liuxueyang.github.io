- [Dominant Character](https://vjudge.net/problem/CodeForces-1605c)

挺有意思的一道贪心题目。官方题解就挺好的，具体就是观察得到有限的符合题意的字符串，发现长度不超过 $7$，所以检查所有长度不大于 $7$ 的子字符串即可。

int n;
string s;

void solve() {
  cin >> n >> s;

  int ca{}, cb{}, cc{}, res = INF;

  For(i, 0, n) {
    int k = min(n, i + 7);
    ca = cb = cc = 0;

    For(j, i, k) {
      if (s[j] == 'a')
        ca++;
      else if (s[j] == 'b')
        cb++;
      else
        cc++;

      if (ca >= 2 && ca > cb && ca > cc) {
        int tmp = j - i + 1;
        ckmin(res, tmp);
      }
    }
  }

  if (res == INF)
    res = -1;
  cout << res << '\n';
}
