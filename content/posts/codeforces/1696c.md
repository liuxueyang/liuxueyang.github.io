- [Fishingprince Plays With Array](https://vjudge.net/problem/CodeForces-1696c)

挺好的一道题目。像这种分久必合，合久必分的题目，我们可以考虑把能分解的都分解。把 $a$ 中的每个元素都分解到不能分解，对 $b$ 中的每个元素，考虑是否可以由分解后 $a$ 的前缀元素组成。有一个更加好写的方法：把 $b$ 中的每个元素也分解到不能分解，然后比较两个分解后的数组是否相等。

有一个陷阱：如果对分解后的 $a$ 数组合并同类项，可能会出现整数溢出，需要使用 `long long` 类型。

const int N = 50100;
int n, m, k, a[N], b[N];

PII split(int x) {
  int cnt = 1;
  while (x % m == 0) {
    cnt *= m;
    x /= m;
  }
  return {x, cnt};
}

using PLL = pair<ll, ll>;

void solve() {
  cin >> n >> m;
  For1(i, 1, n) { cin >> a[i]; }
  cin >> k;
  For1(i, 1, k) { cin >> b[i]; }

  vector<PII> d, d1;

  For1(i, 1, n) { d.pb(split(a[i])); }
  For1(i, 1, k) { d1.pb(split(b[i])); }

  auto shrink = [](vector<PII> &d) {
    vector<PLL> da;
    ll pre = -1, cnt = 0;

    for (auto &[x, c] : d) {
      if (x == pre) {
        cnt += c;
      } else if (pre == -1) {
        pre = x;
        cnt = c;
      } else {
        da.pb({pre, cnt});
        pre = x, cnt = c;
      }
    }
    da.pb({pre, cnt});
    return da;
  };

  vector<PLL> da = shrink(d), db = shrink(d1);

  bool flag = true;

  if (SZ(da) != SZ(db))
    flag = false;
  else {
    int len = SZ(da);
    For(i, 0, len) {
      if (da[i] != db[i]) {
        flag = false;
        break;
      }
    }
  }

  cout << (flag ? "YES" : "NO") << '\n';
}
