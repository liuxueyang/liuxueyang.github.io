- [Turtle Fingers: Count the Values of k](https://codeforces.com/contest/1933/problem/C)

枚举 $a^i$ 中的 $i$，然后枚举 $b^j$，求出对应的 $k$ 即可，由于 $l$ 的范围只有 $10^6$，暴力即可。

void solve() {
  ll a, b, l;
  cin >> a >> b >> l;
  set<int> s;

  int a1 = 1;
  while (a1 <= l) {
    if (l % a1 != 0) {
      a1 *= a;
      continue;
    }

    int l1 = l / a1, b1 = 1;
    while (l1 % b1 == 0) {
      s.insert(l1 / b1);
      b1 *= b;
    }
    a1 *= a;
  }

  cout << SZ(s) << '\n';
}

int main(void) {

  cin >> T;

  while (T--) {
    solve();
  }

  return 0;
