- [Replace With the Previous, Minimize](https://vjudge.net/problem/CodeForces-1675e)

从前往后扫描，记录能够转换到 $a$ 的最大的字符 $c_1$，直到某个字符 $c_i$ 不能转换到 $a$，此时它只能转换到 $c_2 > c_1$。也就是对于字符串中所有 $\le c_1$ 的都转换成 $a$，所有 $[c_2, c_i]$ 范围内的字符，都转换成 $c_2$，其余字符不变。

void solve() {
  int n, k;
  string s;

  cin >> n >> k >> s;
  int delta = 0, c1 = -1, c2 = -1;

  For(i, 0, n) {
    int x = s[i] - 'a';
    if (x <= delta)
      continue;

    if (k >= x - delta) {
      k -= (x - delta);
      delta = x;
      if (!k)
        break;
    } else {
      c1 = x, c2 = x - k;
      break;
    }
  }

  For(i, 0, n) {
    char c3 = 'a' + delta;
    if (delta && s[i] <= c3)
      s[i] = 'a';
    else {
      int j = s[i] - 'a';
      if (c1 != -1 && c2 != -1 && j >= c2 && j <= c1) {
        s[i] = 'a' + c2;
      }
    }
  }

  cout << s << '\n';
}

int main(void) {
