- [Longest Strike](https://vjudge.net/problem/CodeForces-1676f)

找到符合要求的数字，排序之后寻找最长连续的子数组。注意考虑长度是 1 的边界情况。

      cin >> a[i];
      m[a[i]]++;
    }

    for (auto &[x, y] : m) {
      if (y >= k)
        b.pb(x);
    }

          res = cur;
          resl = b[l];
          resr = b[r];
        }
      } else {
        l = i;
      }
    }

    if (res != 0) {
      cout << resl << ' ' << resr << '\n';
    } else {
      cout << "-1\n";
    }
  }

  return 0;
