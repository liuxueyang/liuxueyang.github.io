int a[10], n;
bool ans;

void dfs(int i, int u, int d, int l, int r) {
  if (u < 0 || d < 0 || l < 0 || r < 0 || ans)
    return;

  if (i >= 5 && (u > n - 2 || d > n - 2 || l > n - 2 || r > n - 2)) {
    return;
  }

  if (i >= 5) {
    ans = true;
    return;
  }

  if (i == 1) {
    dfs(i + 1, u - 1, d, l - 1, r);
  } else if (i == 2) {
    dfs(i + 1, u - 1, d, l, r - 1);
  } else if (i == 3) {
    dfs(i + 1, u, d - 1, l - 1, r);
  } else {
    dfs(i + 1, u, d - 1, l, r - 1);
  }
  dfs(i + 1, u, d, l, r);
}

void solve() {
  int u, r, d, l;
  cin >> n >> u >> r >> d >> l;
  ans = false;
