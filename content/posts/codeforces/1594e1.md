- [Rubik's Cube Coloring (easy version)](https://vjudge.net/problem/CodeForces-1594e1)

根节点有 6 种选择，其他节点都有 4 种选择，其他节点的个数是 $2^p - 1 - 1$，所以结果是 $6 \cdot (2^p - 2)$，快速幂。

ll qmi(ll a, ll n, ll m) {
  if (a == 0 || m == 1)
    return 0;

  ll res = 1;
  while (n) {
    if (n & 1)
      res = res * a % m;
    a = a * a % m;
    n >>= 1;
  }
  return res;
