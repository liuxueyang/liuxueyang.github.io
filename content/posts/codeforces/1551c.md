挺有意思的一道贪心题目。

因为字符集只有 $5$，所以可以对每个字母都求一次。假设字符 $c$ 是最终答案，求出它在每个单词中的个数 $cnt_i$，然后求出它对答案的贡献 $cost_i = cnt_i - (len_i - cnt_i)$，根据 $cost_i$ 从大到小排序，前缀和为正的最长的长度就是字符 $c$ 对应的答案。

const int N = 200100;
int n;
string s[N];
int a[N];

int check(char c) {
  For1(i, 1, n) {
    int cnt = 0;
    for (auto x : s[i]) {
      if (x == c)
        cnt++;
    }
    a[i] = cnt - (SZ(s[i]) - cnt);
  }
  sort(a + 1, a + 1 + n, greater<int>());
  int sum = 0;

  For1(i, 1, n) {
    sum += a[i];
    if (sum <= 0)
      return i - 1;
  }

  return n;
}

void solve() {
  cin >> n;

  For1(i, 1, n) { cin >> s[i]; }

  string t = "abcde";
  int ans = 0;

  for (auto c : t) {
    int tmp = check(c);
    ckmax(ans, tmp);
  }

  cout << ans << '\n';
}
