- [Integers Shop](https://vjudge.net/problem/CodeForces-1621B)

只需要考虑最左端点和最右端点，分两种情况：

1. 左右端点各使用一条线段。
2. 用一条线段覆盖最左和最右的端点。

维护左端点的位置 $a$ 和对应的最小价值 $sa$，右端点同理。对于第一种情况，新加入一条线段的时候，更新两个端点的位置和价值。如果两个端点的位置都没有更新过，说明新加入的线段在已有线段的内部，尝试用两个端点的新的价值更新答案。如果两个端点的位置有任何一个更新过，说明范围被扩展了，此时答案直接更新成两个端点的价值和。

对于第二种情况，只需要判定更新之后的端点位置和当前线段是否完全一致即可。

const int N = 100100;

ll n, l[N], r[N], c[N];

void solve() {
  cin >> n;

  ll ans = 0, a, b, sa, sb;

  For1(i, 1, n) {
    cin >> l[i] >> r[i] >> c[i];

    if (i == 1) {
      a = l[i], b = r[i];
      sa = sb = c[i];
      ans = c[i];
      cout << ans << '\n';
      continue;
    }

    bool ext = false;

    if (ckmin(a, l[i])) {
      ext = true;
      sa = c[i];
    }
    if (a == l[i])
      ckmin(sa, c[i]);

    if (ckmax(b, r[i])) {
      ext = true;
      sb = c[i];
    }
    if (b == r[i])
      ckmin(sb, c[i]);

    if (!ext)
      ckmin(ans, sa + sb);
    else
      ans = sa + sb;

    if (l[i] == a && r[i] == b)
      ckmin(ans, c[i]);

    cout << ans << '\n';
  }
}
