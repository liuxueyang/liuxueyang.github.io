- [Phoenix and Towers](https://vjudge.net/problem/CodeForces-1515c)

考虑 $m = 2$ 的情况，容易想到先把方块排序，然后把方块从小到大轮流在两个塔上放是最优的。因此对于 $m$ 个塔，考虑把方块从小到大依次在 $m$ 个塔上放，一直到把方块放完。如果最终最大高度和最小高度的差 $> x$ 说明无解。

const int N = 100100;
int n, m, a[N], x, c[N], d[N];
PII b[N];

void solve() {
  cin >> n >> m >> x;

  For1(i, 1, n) {
    cin >> a[i];
    b[i] = PII(a[i], i);
    c[i] = 0;
  }

  sort(b + 1, b + 1 + n);

  int j = 1;
  For1(i, 1, n) {
    c[j] += b[i].f1;
    d[b[i].f2] = j;
    ++j;
    if (j == m + 1)
      j = 1;
  }

  sort(c + 1, c + 1 + m);
  if (c[m] - c[1] > x) {
    cout << "NO\n";
    return;
  }
  cout << "YES\n";

  cout << '\n';
}
