- [AGAGA XOOORRR](https://vjudge.net/problem/CodeForces-1516b)

求出前缀 $XOR$ 和，$XOR$ 的简单性质，可以求出任意区间 $[l, r]$ 的 $XOR$ 和。

考虑最终局面，有两种情况：1. 只剩下两个数字。2. 只剩下三个数字。其它情况都可以通过操作转化成这两种情况。

因此可以考虑把原数组分成两个区间或者三个区间，而区间的 $XOR$ 和已经得到。

const int N = 2100;

int n, a[N], b[N];

int check(int l, int r) {
  if (l > r)
    return 0;
  return b[r] ^ b[l - 1];
}

void solve() {
  cin >> n;

  b[0] = 0;
  For1(i, 1, n) {
    cin >> a[i];
    b[i] = b[i - 1] ^ a[i];
  }

  bool flag = false;

  For1(j, 2, n) {
    if (flag)
      break;
    For(i, 1, j) {
      int c1 = check(1, i), c2 = check(i + 1, j - 1), c3 = check(j, n);

      if (i + 1 <= j - 1) {
        if (c1 == c2 && c2 == c3) {
          flag = true;
          break;
        }
      } else {
        if (c1 == c3) {
          flag = true;
          break;
        }
      }
    }
  }

  cout << (flag ? "YES" : "NO") << '\n';
}
